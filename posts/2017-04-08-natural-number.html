Haskell/Gofer implementation

data N = Z | S N

lave :: Integer -> N 
lave 0 = Z
lave n = S (lave (n-1))

evalN :: N -> Integer
evalN Z = 0
evalN (S n) = 1+ (evalN n)

foldn (c,h) Z = c
foldn (c,h) (S n) = h (foldn (c,h) n)

plusN m = foldn (m, S) 
multN m = foldn (Z, plusN m)
expnN m = foldn (S Z, multN m)

outl(m,n) = m
outr(m,n) = n

f (m,n) = ( S m, multN n (S m) )
fact = outr . ( foldn((Z,S Z), f) )

g (m,n) = (n, plusN m n)
fib = outl . foldn ((Z,S Z), g)
